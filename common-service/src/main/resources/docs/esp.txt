#include <ESP8266WiFi.h>
#include <ESP8266HTTPClient.h>

//Thay b·∫±ng WiFi th·∫≠t c·ªßa b·∫°n
const char* ssid = "Vo Nhat Truong";
const char* password = "sang0932667080";

//Thay IP v√† port ƒë√∫ng v·ªõi Spring Boot backend
const char* serverUrl = "http://192.168.1.118:9997/api/v1/sensors";

const float U0 = 2000.0;
float Um;
float f;
int NTU;
long ssvs = 0;

#define relay 12  // Ch√¢n D6

void setup() {
  Serial.begin(9600);
  pinMode(relay, OUTPUT);
  digitalWrite(relay, LOW);  // M·∫∑c ƒë·ªãnh t·∫Øt relay

  // K·∫øt n·ªëi WiFi
  WiFi.begin(ssid, password);
  Serial.print("üîå ƒêang k·∫øt n·ªëi WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\n‚úÖ ƒê√£ k·∫øt n·ªëi WiFi");
}

void loop() {
  // L·∫•y m·∫´u c·∫£m bi·∫øn
  for (int i = 0; i < 100; i++) {
    int sensorValue = analogRead(A0);
    ssvs += sensorValue;
  }

  int ssvt = ssvs / 100;
  ssvs = 0;

  Um = (ssvt / 1023.0) * 3300;
  f = Um / U0;

  if (f >= 0.98) {
    NTU = 0;
  } else {
    NTU = map(f * 100, 0, 100, 1000, 0);
  }

  // ƒêi·ªÅu khi·ªÉn relay
  bool isClear = NTU < 150;
  digitalWrite(relay, isClear ? HIGH : LOW);

  // In ra Serial
  Serial.print("Um (mV): ");
  Serial.print(Um);
  Serial.print("\t f: ");
  Serial.print(f, 4);
  Serial.print("\t NTU: ");
  Serial.print(NTU);
  Serial.print("\t ");
  Serial.print(isClear ? "N∆∞·ªõc trong" : "N∆∞·ªõc ƒë·ª•c");
  Serial.print("\t Relay ");
  Serial.println(isClear ? "M·ªû" : "ƒê√ìNG");

  // G·ª≠i d·ªØ li·ªáu l√™n server
  if (WiFi.status() == WL_CONNECTED) {
    WiFiClient client;
    HTTPClient http;
    http.begin(client, serverUrl);  // ‚úÖ D√πng phi√™n b·∫£n ƒë√∫ng: http.begin(client, url)
    http.addHeader("Content-Type", "application/json");

    long timestamp = millis();  // S·ª≠ d·ª•ng th·ªùi gian t·ª´ millis()
    String jsonPayload = "{";
    jsonPayload += "\"turbidity\":" + String(NTU) + ",";
    jsonPayload += "\"timestamp\":" + String(timestamp);
    jsonPayload += "}";

    int httpCode = http.POST(jsonPayload);
    if (httpCode > 0) {
      Serial.println("üì° G·ª≠i d·ªØ li·ªáu th√†nh c√¥ng");
    } else {
      Serial.print("‚ùå L·ªói g·ª≠i d·ªØ li·ªáu. M√£ l·ªói: ");
      Serial.println(httpCode);
    }

    http.end();
  } else {
    Serial.println("‚ö†Ô∏è WiFi m·∫•t k·∫øt n·ªëi, kh√¥ng th·ªÉ g·ª≠i d·ªØ li·ªáu.");
  }

  delay(5000);  // L·∫∑p m·ªói 5 gi√¢y
}
